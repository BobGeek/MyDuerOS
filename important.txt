安装pip,setuptools
wget https://bootstrap.pypa.io/get-pip.py
sudo python get-pip.py
安装Python2.7.14
wget https://www.python.org/ftp/python/2.7.14/Python-2.7.14.tgz
tar -zxvf Python-2.7.14.tgz
cd Python-2.7.14/
./configure
make
sudo make install
